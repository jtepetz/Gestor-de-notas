INICIO
    // GESTOR DE NOTAS ACADÉMICAS
    // Autor: Josué Elías Tepet Zet
    // Descripción: Sistema en consola para registrar, visualizar y analizar cursos y notas.
    // Este pseudocódigo describe el funcionamiento general del programa principal.

    // Declaración de estructuras principales
    cursos ← lista vacía
    historial_cambios ← lista vacía    // (pila)
    cola_revision ← lista vacía         // (cola)

    // MENÚ PRINCIPAL
    MIENTRAS VERDADERO HACER
        IMPRIMIR "====== GESTOR DE NOTAS ACADÉMICAS ======"
        IMPRIMIR "1. Registrar nuevo curso"
        IMPRIMIR "2. Mostrar todos los cursos y notas"
        IMPRIMIR "3. Calcular promedio general"
        IMPRIMIR "4. Contar cursos aprobados y reprobados"
        IMPRIMIR "5. Buscar curso por nombre (búsqueda lineal)"
        IMPRIMIR "6. Actualizar nota de un curso"
        IMPRIMIR "7. Eliminar un curso"
        IMPRIMIR "8. Ordenar cursos por nota (burbuja)"
        IMPRIMIR "9. Ordenar cursos por nombre (inserción)"
        IMPRIMIR "10. Buscar curso por nombre (búsqueda binaria)"
        IMPRIMIR "11. Simular cola de solicitudes de revisión"
        IMPRIMIR "12. Mostrar historial de cambios (pila)"
        IMPRIMIR "13. Salir"
        IMPRIMIR "Seleccione una opción: "
        LEER opcion

        // OPCIONES DEL MENÚ
        SI opcion = 1 ENTONCES
            IMPRIMIR "Ingrese nombre del curso y nota."
            LEER nombre, nota
            SI nombre ≠ "" Y nota ENTRE 0 Y 100 ENTONCES
                cursos.AGREGAR({nombre, nota})
                IMPRIMIR "Curso registrado correctamente."
            SINO
                IMPRIMIR "Datos inválidos, intente de nuevo."
            FIN_SI

        SINO SI opcion = 2 ENTONCES
            SI cursos.VACÍO ENTONCES
                IMPRIMIR "No hay cursos registrados."
            SINO
                PARA i ← 1 HASTA cursos.LONGITUD HACER
                    IMPRIMIR i, cursos[i].nombre, " - Nota: ", cursos[i].nota
                FIN_PARA
            FIN_SI

        SINO SI opcion = 3 ENTONCES
            SI cursos.VACÍO ENTONCES
                IMPRIMIR "No hay cursos registrados."
            SINO
                suma ← 0
                PARA cada curso EN cursos HACER
                    suma ← suma + curso.nota
                FIN_PARA
                promedio ← suma / cursos.LONGITUD
                IMPRIMIR "Promedio general: ", promedio
            FIN_SI

        SINO SI opcion = 4 ENTONCES
            aprobados ← 0
            reprobados ← 0
            PARA cada curso EN cursos HACER
                SI curso.nota >= 60 ENTONCES
                    aprobados ← aprobados + 1
                SINO
                    reprobados ← reprobados + 1
                FIN_SI
            FIN_PARA
            IMPRIMIR "Aprobados: ", aprobados
            IMPRIMIR "Reprobados: ", reprobados

        SINO SI opcion = 5 ENTONCES
            IMPRIMIR "Ingrese nombre del curso a buscar:"
            LEER nombre_buscar
            encontrado ← FALSO
            PARA cada curso EN cursos HACER
                SI nombre_buscar EN curso.nombre ENTONCES
                    IMPRIMIR "Curso encontrado: ", curso.nombre, " - Nota: ", curso.nota
                    encontrado ← VERDADERO
                FIN_SI
            FIN_PARA
            SI NO encontrado ENTONCES
                IMPRIMIR "Curso no encontrado."
            FIN_SI

        SINO SI opcion = 6 ENTONCES
            LEER nombre
            PARA cada curso EN cursos HACER
                SI curso.nombre = nombre ENTONCES
                    LEER nueva_nota
                    historial_cambios.APILAR("Actualización de " + nombre)
                    curso.nota ← nueva_nota
                    IMPRIMIR "Nota actualizada."
                FIN_SI
            FIN_PARA

        SINO SI opcion = 7 ENTONCES
            LEER nombre
            PARA cada curso EN cursos HACER
                SI curso.nombre = nombre ENTONCES
                    IMPRIMIR "¿Desea eliminar este curso? (s/n)"
                    LEER confirmar
                    SI confirmar = "s" ENTONCES
                        historial_cambios.APILAR("Eliminación de " + nombre)
                        cursos.ELIMINAR(curso)
                        IMPRIMIR "Curso eliminado."
                    FIN_SI
                FIN_SI
            FIN_PARA

        SINO SI opcion = 8 ENTONCES
            // ORDENAMIENTO BURBUJA POR NOTA (descendente)
            PARA i ← 0 HASTA cursos.LONGITUD - 2 HACER
                PARA j ← 0 HASTA cursos.LONGITUD - i - 2 HACER
                    SI cursos[j].nota < cursos[j+1].nota ENTONCES
                        INTERCAMBIAR cursos[j], cursos[j+1]
                    FIN_SI
                FIN_PARA
            FIN_PARA
            IMPRIMIR "Cursos ordenados por nota (descendente)."

        SINO SI opcion = 9 ENTONCES
            // ORDENAMIENTO POR INSERCIÓN (alfabético)
            PARA i ← 1 HASTA cursos.LONGITUD - 1 HACER
                actual ← cursos[i]
                j ← i - 1
                MIENTRAS j >= 0 Y cursos[j].nombre > actual.nombre HACER
                    cursos[j + 1] ← cursos[j]
                    j ← j - 1
                FIN_MIENTRAS
                cursos[j + 1] ← actual
            FIN_PARA
            IMPRIMIR "Cursos ordenados por nombre."

        SINO SI opcion = 10 ENTONCES
            // BÚSQUEDA BINARIA (requiere lista ordenada por nombre)
            ordenar_por_nombre()
            LEER nombre_buscar
            inicio ← 0
            fin ← cursos.LONGITUD - 1
            encontrado ← FALSO
            MIENTRAS inicio <= fin Y NO encontrado HACER
                medio ← (inicio + fin) / 2
                SI cursos[medio].nombre = nombre_buscar ENTONCES
                    IMPRIMIR "Curso encontrado: ", cursos[medio].nombre, " - Nota: ", cursos[medio].nota
                    encontrado ← VERDADERO
                SINO SI cursos[medio].nombre < nombre_buscar ENTONCES
                    inicio ← medio + 1
                SINO
                    fin ← medio - 1
                FIN_SI
            FIN_MIENTRAS
            SI NO encontrado ENTONCES
                IMPRIMIR "Curso no encontrado."
            FIN_SI

        SINO SI opcion = 11 ENTONCES
            IMPRIMIR "Ingrese cursos para revisión (escriba 'fin' para terminar):"
            LEER curso
            MIENTRAS curso ≠ "fin" HACER
                cola_revision.ENCOLAR(curso)
                LEER curso
            FIN_MIENTRAS
            MIENTRAS cola_revision.NO_VACÍA() HACER
                actual ← cola_revision.DESCOLAR()
                IMPRIMIR "Revisando: ", actual
            FIN_MIENTRAS
            IMPRIMIR "Solicitudes procesadas."

        SINO SI opcion = 12 ENTONCES
            SI historial_cambios.VACÍO ENTONCES
                IMPRIMIR "No hay historial."
            SINO
                IMPRIMIR "Historial de cambios (más reciente primero):"
                MIENTRAS historial_cambios.NO_VACÍA() HACER
                    cambio ← historial_cambios.DESAPILAR()
                    IMPRIMIR cambio
                FIN_MIENTRAS
            FIN_SI

        SINO SI opcion = 13 ENTONCES
            IMPRIMIR "Gracias por usar el Gestor de Notas Académicas. ¡Hasta pronto!"
            SALIR DEL BUCLE

        SINO
            IMPRIMIR "Opción inválida. Intente nuevamente."
        FIN_SI

    FIN_MIENTRAS

FIN
